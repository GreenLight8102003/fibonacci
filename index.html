    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

def isFibonacci(x, n=0):
    # Điều kiện dừng: nếu tìm thấy số Fibonacci bằng x
    fib = fibonacci(n)
    if fib == x:
        return True
    # Nếu số Fibonacci vượt quá x, dừng tìm kiếm
    if fib > x:
        return False
    # Đệ quy: kiểm tra số Fibonacci tiếp theo
    return isFibonacci(x, n + 1)
int fibonacci(int n) {

bool isFibonacci(int x, int n = 0) {
    // Điều kiện dừng: nếu tìm thấy số Fibonacci bằng x
    int fib = fibonacci(n);
    if (fib == x)
        return true;
    // Nếu số Fibonacci vượt quá x, dừng tìm kiếm
    if (fib > x)
        return false;
    # Đệ quy: kiểm tra số Fibonacci tiếp theo
    return isFibonacci(x, n + 1);
}
